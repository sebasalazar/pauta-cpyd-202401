108double utem::calculateMedian(std::vector<double> list) {
109 // Obtener el iterador al elemento mediano
110 std::vector<double>::const_iterator first = list.begin();
111 std::advance(first, list.size() / 2);
112
113 // Calcular la mediana
114 if (list.size() % 2 == 0) {
115 // Tamaño par: mediana es el promedio de los dos valores centrales
116 std::vector<double>::const_iterator second = first;
117 --second; // Retroceder un paso para el segundo valor central
118 return (*first + *second) / 2.0;
119 } else {
120 // Tamaño impar: mediana es el valor central
121 return *first;
